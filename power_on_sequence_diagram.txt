@startuml

title "Power On"

actor User
boundary "Power button" as powerkey
boundary GUI
control OS
control StartupManager
control "DICOM service" as dicom
control CameraManager

User -> powerkey : press
powerkey -> OS : key event
OS -> StartupManager: launch
activate StartupManager
StartupManager -> GUI : boot animation
StartupManager -> dicom : activate
activate dicom
alt initialization success
  StartupManager <- dicom : success
else initiaization failure
  StartupManager <- dicom : fail
  StartupManager -> GUI : error message
  GUI -> User
end
StartupManager ->] : "Self-Test"
alt Self-Test success
  StartupManager <-] : success
else some error occurred
  StartupManager <-] : fail
  StartupManager -> GUI : error message
  GUI -> User
end
StartupManager -> CameraManager : enable live view
StartupManager -> GUI : activate user interface


@enduml
