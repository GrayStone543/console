usecase "Power on"

actor User
boundary "Power button" as powerkey
boundary Panel
boundary LED
control LiveViewManager #lightgreen
entity Console #cyan

powerkey --> Console : "key event"
Console --> LiveViewManager #line:blue;text:blue : "start live view"
LiveViewManager --> Panel #line:orange;text:orange : "Live images"
Console --> Panel #line:blue : "turn on"
Console --> LED #line:blue : "turn on"
Panel --> User #line:green;text:green : "display"
LED --> User #line:green;text:green : "light indication"

@enduml
